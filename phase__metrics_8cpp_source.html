148 void PhaseMetrics::MergeMetrics(PhaseMetrics& metrics)
149 {
150  // Choose best min time
151  if (metrics._min_time < _min_time)
152  _min_time = metrics._min_time;
153 
154  // Choose best max time

157

160  _custom_uint.insert(metrics._custom_uint.begin(), metrics._custom_uint.end());
161  _custom_int64.insert(metrics._custom_int64.begin(), metrics._custom_int64.end());
162  _custom_uint64.insert(metrics._custom_uint64.begin(), metrics._custom_uint64.end());
163  _custom_flt.insert(metrics._custom_flt.begin(), metrics._custom_flt.end());
164  _custom_dbl.insert(metrics._custom_dbl.begin(), metrics._custom_dbl.end());
165  _custom_str.insert(metrics._custom_str.begin(), metrics._custom_str.end());
166 
167  // Choose best total time with iterations, items and bytes
168  if (metrics._total_time < _total_time)
169  {
170  std::swap(_latency, metrics._latency);
171  _total_time = metrics._total_time;
172  _total_iterations = metrics._total_iterations;
173  _total_items = metrics._total_items;
174  _total_bytes = metrics._total_bytes;
175  // Overwrite metrics custom tables
176  for (auto& it : metrics._custom_int)
177  _custom_int[it.first] = it.second;
178  for (auto& it : metrics._custom_uint)
179  _custom_uint[it.first] = it.second;
180  for (auto& it : metrics._custom_int64)
181  _custom_int64[it.first] = it.second;
182  for (auto& it : metrics._custom_uint64)
183  _custom_uint64[it.first] = it.second;
184  for (auto& it : metrics._custom_flt)
185  _custom_flt[it.first] = it.second;
186  for (auto& it : metrics._custom_dbl)
187  _custom_dbl[it.first] = it.second;
188  for (auto& it : metrics._custom_str)

190  // Overwrite metrics threads value
